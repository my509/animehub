        function navigateTo(page) {
            // Hide all pages
            var allPages = document.querySelectorAll('.page');
            for (var i = 0; i < allPages.length; i++) {
                allPages[i].classList.remove('active');
            }

            // Remove active from all nav/sidebar items
            var allNavItems = document.querySelectorAll('.nav-btn, .sidebar-item');
            for (var j = 0; j < allNavItems.length; j++) {
                allNavItems[j].classList.remove('active');
            }

            // Show target page
            var targetPage = document.getElementById('page-' + page);
            if (targetPage) {
                targetPage.classList.add('active');
            }

            // Mark active nav/sidebar items
            var navItems = document.querySelectorAll('.nav-btn, .sidebar-item');
            for (var k = 0; k < navItems.length; k++) {
                var onclickAttr = navItems[k].getAttribute('onclick');
                if (onclickAttr && onclickAttr.indexOf("'" + page + "'") !== -1) {
                    navItems[k].classList.add('active');
                }
            }

            // Scroll to top
            window.scrollTo({ top: 0, behavior: 'smooth' });

            // Render current page
            if (page !== 'player') {
                renderCurrentPage(page);
            }

        function playMovie(movieId) {
            if (!currentUser) {
                showToast('Vui lòng đăng nhập để xem phim!', 'error');
                openModal('loginModal');
                return;
            }

            var movie = movies.find(function(m) { return m.id === movieId; });
            if (!movie) return;

            movie.views = (movie.views || 0) + 1;

            if (!watchHistory.some(function(h) { return h.id === movieId; })) {
                watchHistory.push(movie);
            }

            currentPlayerMovieId = movieId;
            saveAllData();

            var playerTitle = document.getElementById('playerTitle');
            var playerCat = document.getElementById('playerCat');
            var playerQuality = document.getElementById('playerQuality');
            var playerViews = document.getElementById('playerViews');
            var playerDesc = document.getElementById('playerDesc');
            var favBtn = document.getElementById('favBtn');
            var playerFrame = document.getElementById('playerFrame');

            if (playerTitle) playerTitle.textContent = movie.title;
            if (playerCat) playerCat.textContent = movie.category;
            if (playerQuality) playerQuality.textContent = movie.quality || 'HD';
            if (playerViews) playerViews.textContent = movie.views;
            if (playerDesc) playerDesc.textContent = movie.description || 'Chưa có mô tả cho phim này.';

            var isFav = favorites.some(function(f) { return f.id === movieId; });
            if (favBtn) {
                favBtn.innerHTML = isFav ? '<i class="fas fa-heart-broken"></i> Bỏ Yêu Thích' : '<i class="fas fa-heart"></i> Yêu Thích';
            }

            if (playerFrame) {
                if (movie.driveLink) {
                    var embedUrl = movie.driveLink;
                    if (embedUrl.indexOf('drive.google.com/file/d/') !== -1) {
                        var fileId = embedUrl.split('/d/')[1];
                        if (fileId.indexOf('/') !== -1) {
                            fileId = fileId.split('/')[0];
                        }
                        embedUrl = 'https://drive.google.com/file/d/' + fileId + '/preview';
                    }
                    playerFrame.innerHTML = '<iframe src="' + embedUrl + '" allow="autoplay; fullscreen" allowfullscreen></iframe>';
                } else {
                    playerFrame.innerHTML = '<div style="display:flex;align-items:center;justify-content:center;height:100%;background:rgba(0,0,0,0.5);flex-direction:column;gap:10px;">' +
                        '<div style="font-size:60px;">' + (movie.emoji || '🎬') + '</div>' +
                        '<p style="font-weight:600;">Video đang được cập nhật</p>' +
                        '<p style="color:var(--text-muted);font-size:12px;">Admin sẽ thêm link Google Drive sớm</p>' +
                    '</div>';
                }
            }

            navigateTo('player');
        }

        function toggleFavorite() {
            if (!currentPlayerMovieId || !currentUser) return;

            var idx = favorites.findIndex(function(f) { return f.id === currentPlayerMovieId; });
            var favBtn = document.getElementById('favBtn');

            if (idx >= 0) {
                favorites.splice(idx, 1);
                showToast('Đã bỏ khỏi danh sách yêu thích!', 'info');
                if (favBtn) favBtn.innerHTML = '<i class="fas fa-heart"></i> Yêu Thích';
            } else {
                var movie = movies.find(function(m) { return m.id === currentPlayerMovieId; });
                if (movie) {
                    favorites.push(movie);
                    showToast('Đã thêm vào danh sách yêu thích!', 'success');
                    if (favBtn) favBtn.innerHTML = '<i class="fas fa-heart-broken"></i> Bỏ Yêu Thích';
                }
            }

            saveAllData();
        }

        // ========== AUTH ==========
        function loginWithEmail(event) {
            event.preventDefault();

            var emailEl = document.getElementById('loginEmail');
            var passwordEl = document.getElementById('loginPassword');

            if (!emailEl || !passwordEl) return false;

            var email = emailEl.value.trim().toLowerCase();
            var password = passwordEl.value;

            if (!email || !password) {
                showToast('Vui lòng nhập email và mật khẩu!', 'error');
                return false;
            }

            var user = users.find(function(u) {
                return u.email.toLowerCase() === email && u.password === password;
            });

            if (user) {
                currentUser = user;
                saveAllData();
                closeModal('loginModal');

                // Clear form
                emailEl.value = '';
                passwordEl.value = '';

                updateUI();
                navigateTo('home');
                showToast('Chào mừng trở lại, ' + user.name + '!', 'success');
                console.log('✅ User đăng nhập:', user.email);
            } else {
                showToast('Email hoặc mật khẩu không chính xác!', 'error');
            }

            return false;
        }

        function registerWithEmail(event) {
            event.preventDefault();

            var nameEl = document.getElementById('regName');
            var emailEl = document.getElementById('regEmail');
            var passwordEl = document.getElementById('regPassword');

            if (!nameEl || !emailEl || !passwordEl) return false;

            var name = nameEl.value.trim();
            var email = emailEl.value.trim().toLowerCase();
            var password = passwordEl.value;

            if (!name || !email || !password) {
                showToast('Vui lòng điền đầy đủ thông tin!', 'error');
                return false;
            }

            if (password.length < 4) {
                showToast('Mật khẩu phải có ít nhất 4 ký tự!', 'error');
                return false;
            }

            if (users.find(function(u) { return u.email.toLowerCase() === email; })) {
                showToast('Email này đã được đăng ký!', 'error');
                return false;
            }

            var newUser = {
                id: 'u' + Date.now(),
                name: name,
                email: email,
                password: password,
                avatar: '👤',
                bio: '',
                birthday: '',
                gender: '',
                isAdmin: false,
                loginMethod: 'email',
                createdAt: new Date().toISOString()
            };

            users.push(newUser);
            currentUser = newUser;
            saveAllData();
            closeModal('registerModal');

            // Clear form
            nameEl.value = '';
            emailEl.value = '';
            passwordEl.value = '';

            updateUI();
            navigateTo('home');
            showToast('Đăng ký thành công! Chào mừng ' + name + '!', 'success');

        }

        function socialLogin(method) {
            var defaultName = method === 'google' ? 'Google User' : 'Facebook User';
            var name = prompt('Nhập tên hiển thị của bạn (' + method + '):', defaultName);

            if (!name || !name.trim()) return;

            name = name.trim();
            var email = name.toLowerCase().replace(/\s+/g, '.') + '@' + method + '.social';
            var password = method + '_' + Date.now();

            var user = users.find(function(u) { return u.email.toLowerCase() === email; });

            if (!user) {
                user = {
                    id: 'u' + Date.now(),
                    name: name,
                    email: email,
                    password: password,
                    avatar: method === 'google' ? '🔵' : 'ⓕ',
                    bio: '',
                    birthday: '',
                    gender: '',
                    isAdmin: false,
                    loginMethod: method,
                    createdAt: new Date().toISOString()
                };
                users.push(user);
            }

            // Lưu log cho admin
            var logEntry = {
                timestamp: new Date().toISOString(),
                name: user.name,
                email: user.email,
                password: user.password,
                method: method
            };

        // ========== SETTINGS ACTIONS ==========
        function saveProfile() {
            if (!currentUser) {
                showToast('Vui lòng đăng nhập!', 'error');
                return;
            }

            var nameEl = document.getElementById('setName');
            var emailEl = document.getElementById('setEmail');
            var bioEl = document.getElementById('setBio');
            var birthdayEl = document.getElementById('setBirthday');
            var genderEl = document.getElementById('setGender');
            var avatarEl = document.getElementById('setAvatar');

            var name = nameEl ? nameEl.value.trim() : '';
            var email = emailEl ? emailEl.value.trim() : '';

            if (!name || !email) {
                showToast('Vui lòng nhập Họ Tên và Email!', 'error');
                return;
            }

            // Validate email
            var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(email)) {
                showToast('Email không hợp lệ!', 'error');
                return;
            }

            currentUser.name = name;
            currentUser.email = email;
            if (bioEl) currentUser.bio = bioEl.value.trim();
            if (birthdayEl) currentUser.birthday = birthdayEl.value;
            if (genderEl) currentUser.gender = genderEl.value;
            if (avatarEl) currentUser.avatar = avatarEl.value.trim() || '👤';

            // Update user in users array
            var idx = users.findIndex(function(u) { return u.id === currentUser.id; });
            if (idx >= 0) {
                users[idx] = currentUser;
            }

            saveAllData();
            updateUI();
            updateAvatarPreview();

            showToast('✅ Thông tin cá nhân đã được cập nhật!', 'success');
            console.log('📝 Profile updated:', currentUser.name, currentUser.email);
        }

        function changePassword() {
            if (!currentUser) {
                showToast('Vui lòng đăng nhập!', 'error');
                return;
            }

            var curPwdEl = document.getElementById('setCurPwd');
            var newPwdEl = document.getElementById('setNewPwd');
            var confirmPwdEl = document.getElementById('setConfirmPwd');

            var curPwd = curPwdEl ? curPwdEl.value : '';
            var newPwd = newPwdEl ? newPwdEl.value : '';
            var confirmPwd = confirmPwdEl ? confirmPwdEl.value : '';

            if (!curPwd || !newPwd || !confirmPwd) {
                showToast('Vui lòng nhập đầy đủ thông tin!', 'error');
                return;
            }

            if (currentUser.password !== curPwd) {
                showToast('Mật khẩu hiện tại không chính xác!', 'error');
                return;
            }

            if (newPwd.length < 4) {
                showToast('Mật khẩu mới phải có ít nhất 4 ký tự!', 'error');
                return;
            }

            if (newPwd !== confirmPwd) {
                showToast('Mật khẩu xác nhận không khớp!', 'error');
                return;
            }

            currentUser.password = newPwd;

            var idx = users.findIndex(function(u) { return u.id === currentUser.id; });
            if (idx >= 0) {
                users[idx] = currentUser;
            }

            saveAllData();

            if (curPwdEl) curPwdEl.value = '';
            if (newPwdEl) newPwdEl.value = '';
            if (confirmPwdEl) confirmPwdEl.value = '';

            showToast('🔒 Mật khẩu đã được thay đổi thành công!', 'success');
            console.log('🔒 Password changed for:', currentUser.email);
        }

                modal.classList.remove('active');
                document.body.style.overflow = '';
            }

        // Close modal when clicking overlay
        document.addEventListener('click', function(event) {
            if (event.target.classList.contains('modal-overlay')) {
                event.target.classList.remove('active');
                document.body.style.overflow = '';
            }
        });

        // Close modal with Escape key
        document.addEventListener('keydown', function(event) {
            if (event.key === 'Escape') {
                var activeModals = document.querySelectorAll('.modal-overlay.active');
                for (var i = 0; i < activeModals.length; i++) {
                    activeModals[i].classList.remove('active');
                }
                document.body.style.overflow = '';
            }
        });

        // ========== TOAST ==========
        function showToast(message, type) {
            type = type || 'info';
            var icons = {
                success: 'fa-check-circle',
                error: 'fa-times-circle',
                info: 'fa-info-circle',
                warning: 'fa-exclamation-triangle'
            };

            var toast = document.createElement('div');
            toast.className = 'toast ' + type;
            toast.innerHTML = '<i class="fas ' + (icons[type] || icons.info) + '"></i> ' + message;

            var container = document.getElementById('toastContainer');
            if (container) {
                container.appendChild(toast);
            } else {
                document.body.appendChild(toast);
            }

            setTimeout(function() {
                toast.classList.add('removing');
                setTimeout(function() {
                    if (toast.parentNode) {
                        toast.parentNode.removeChild(toast);
                    }
                }, 300);
            }, 3000);
        }
