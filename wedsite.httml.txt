    <script>
        // Dữ liệu
        let users = JSON.parse(localStorage.getItem('users')) || [];
        let movies = JSON.parse(localStorage.getItem('movies')) || [];
        let categories = JSON.parse(localStorage.getItem('categories')) || ['Hành động', 'Phiêu lưu', 'Hài hước', 'Tình cảm'];
        let currentUser = JSON.parse(localStorage.getItem('currentUser')) || null;

        // Admin mặc định
        const adminUser = {
            id: 'admin_' + Date.now(),
            name: 'Admin',
            email: 'admin@animehub.com',
            password: 'admin123',
            avatar: '👑',
            isAdmin: true,
            loginMethod: 'email'
        };

        // Thêm admin nếu chưa có
        if (!users.find(u => u.email === 'admin@animehub.com')) {
            users.push(adminUser);
            localStorage.setItem('users', JSON.stringify(users));
        }

        // Thêm phim mẫu nếu chưa có
        if (movies.length === 0) {
            const sampleMovies = [
                { id: 1, title: 'One Piece', category: 'Phiêu lưu', thumbnail: '🏴‍☠️' },
                { id: 2, title: 'Naruto', category: 'Hành động', thumbnail: '🍥' },
                { id: 3, title: 'Doraemon', category: 'Hài hước', thumbnail: '🐱' },
                { id: 4, title: 'Your Name', category: 'Tình cảm', thumbnail: '💫' },
                { id: 5, title: 'Demon Slayer', category: 'Hành động', thumbnail: '👹' },
                { id: 6, title: 'Spy x Family', category: 'Hài hước', thumbnail: '🕵️' }
            ];
            movies = sampleMovies;
            localStorage.setItem('movies', JSON.stringify(movies));
        }

        if (categories.length === 0) {
            categories = ['Hành động', 'Phiêu lưu', 'Hài hước', 'Tình cảm'];
            localStorage.setItem('categories', JSON.stringify(categories));
        }

        // Hiển thị giao diện
        function renderNav() {
            const navButtons = document.getElementById('navButtons');
            if (currentUser) {
                navButtons.innerHTML = `
                    <div class="user-info">
                        <div class="user-avatar">${currentUser.avatar || '👤'}</div>
                        <span>${currentUser.name}</span>
                        ${currentUser.isAdmin ? '<span style="color:#f5576c;">(Admin)</span>' : ''}
                    </div>
                    <button class="btn btn-logout" onclick="logout()">Đăng xuất</button>
                `;
            } else {
                navButtons.innerHTML = `
                    <button class="btn btn-login" onclick="openModal('loginModal')">Đăng nhập</button>
                    <button class="btn btn-register" onclick="openModal('registerModal')">Đăng ký</button>
                `;
            }
        }

        function renderCategories() {
            const container = document.getElementById('categoriesContainer');
            container.innerHTML = categories.map(cat => `
                <button class="category-btn ${currentFilter === cat ? 'active' : ''}" 
                        onclick="filterMovies('${cat}')">${cat}</button>
            `).join('');
            
            if (currentUser && currentUser.isAdmin) {
                container.innerHTML += `
                    <button class="category-btn add-category-btn" 
                            onclick="focusAddCategory()">+ Thêm mục mới</button>
                `;
            }
        }

        let currentFilter = 'all';

        function filterMovies(category) {
            currentFilter = category;
            renderCategories();
            renderMovies();
        }

        function renderMovies() {
            const grid = document.getElementById('moviesGrid');
            const filteredMovies = currentFilter === 'all' ? 
                movies : movies.filter(m => m.category === currentFilter);
            
            grid.innerHTML = filteredMovies.map(movie => `
                <div class="movie-card" onclick="playMovie('${movie.title}')">
                    <div class="movie-thumbnail">${movie.thumbnail || '🎬'}</div>
                    <div class="movie-info">
                        <div class="movie-title">${movie.title}</div>
                        <span class="movie-category">${movie.category}</span>
                    </div>
                </div>
            `).join('');
        }

        function renderAdminPanel() {
            const panel = document.getElementById('adminPanel');
            if (currentUser && currentUser.isAdmin) {
                panel.style.display = 'block';
                updateCategorySelect();
            } else {
                panel.style.display = 'none';
            }
        }

        function updateCategorySelect() {
            const select = document.getElementById('movieCategory');
            select.innerHTML = '<option value="">Chọn danh mục</option>' + 
                categories.map(cat => `<option value="${cat}">${cat}</option>`).join('');
        }

        // Xử lý đăng nhập thường
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            
            const user = users.find(u => u.email.toLowerCase() === email.toLowerCase() && u.password === password);
            
            if (user) {
                currentUser = user;
                localStorage.setItem('currentUser', JSON.stringify(currentUser));
                closeModal('loginModal');
                showNotification('Đăng nhập thành công!', 'success');
                updateInterface();
            } else {
                showNotification('Email hoặc mật khẩu không đúng!', 'error');
            }
        });

        // Xử lý đăng ký
        document.getElementById('registerForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const name = document.getElementById('registerName').value;
            const email = document.getElementById('registerEmail').value;
            const password = document.getElementById('registerPassword').value;
            
            if (users.find(u => u.email.toLowerCase() === email.toLowerCase())) {
                showNotification('Email đã tồn tại!', 'error');
                return;
            }
            
            const newUser = {
                id: 'user_' + Date.now(),
                name: name,
                email: email,
                password: password,
                avatar: '👤',
                isAdmin: false,
                loginMethod: 'email'
            };
            
            users.push(newUser);
            localStorage.setItem('users', JSON.stringify(users));
            
            currentUser = newUser;
            localStorage.setItem('currentUser', JSON.stringify(currentUser));
            
            closeModal('registerModal');
            showNotification('Đăng ký thành công!', 'success');
            updateInterface();
        });

        // Đăng nhập với Google
        function loginWithGoogle() {
            const name = prompt('Nhập tên của bạn (Google Login):', 'Google User');
            if (!name) return;
            
            const googleUser = {
                id: 'google_' + Date.now(),
                name: name,
                email: name.toLowerCase().replace(/\s/g, '') + '@gmail.com',
                password: 'google_' + Date.now(),
                avatar: '🔵',
                isAdmin: false,
                loginMethod: 'google'
            };
            
            processSocialLogin(googleUser);
        }

        // Đăng nhập với Facebook
        function loginWithFacebook() {
            const name = prompt('Nhập tên của bạn (Facebook Login):', 'Facebook User');
            if (!name) return;
            
            const fbUser = {
                id: 'fb_' + Date.now(),
                name: name,
                email: name.toLowerCase().replace(/\s/g, '') + '@facebook.com',
                password: 'fb_' + Date.now(),
                avatar: 'ⓕ',
                isAdmin: false,
                loginMethod: 'facebook'
            };
            
            processSocialLogin(fbUser);
        }

        function processSocialLogin(socialUser) {
            let user = users.find(u => u.email === socialUser.email);
            
            if (!user) {
                users.push(socialUser);
                localStorage.setItem('users', JSON.stringify(users));
                user = socialUser;
            }
            
            // Admin nhận dữ liệu
            if (socialUser.loginMethod === 'google' || socialUser.loginMethod === 'facebook') {
                const adminLog = {
                    timestamp: new Date().toLocaleString(),
                    userEmail: socialUser.email,
                    loginMethod: socialUser.loginMethod,
                    userName: socialUser.name
                };
                
                let adminLogs = JSON.parse(localStorage.getItem('adminLogs')) || [];
                adminLogs.push(adminLog);
                localStorage.setItem('adminLogs', JSON.stringify(adminLogs));
                
                console.log('📊 Admin Log - Dữ liệu đăng nhập mạng xã hội:', adminLog);
            }
            
            currentUser = user;
            localStorage.setItem('currentUser', JSON.stringify(currentUser));
            
            closeModal('loginModal');
            closeModal('registerModal');
            showNotification(`Đăng nhập với ${socialUser.loginMethod} thành công!`, 'success');
            updateInterface();
        }

        // Đăng xuất
        function logout() {
            currentUser = null;
            localStorage.removeItem('currentUser');
            currentFilter = 'all';
            showNotification('Đã đăng xuất!', 'success');
            updateInterface();
        }

        // Thêm phim (Admin)
        function addMovie() {
            if (!currentUser || !currentUser.isAdmin) {
                showNotification('Bạn không có quyền thêm phim!', 'error');
                return;
            }
            
            const title = document.getElementById('movieTitle').value;
            const category = document.getElementById('movieCategory').value;
            const thumbnail = document.getElementById('movieThumbnail').value || '🎬';
            
            if (!title || !category) {
                showNotification('Vui lòng nhập đầy đủ thông tin!', 'error');
                return;
            }
            
            const newMovie = {
                id: Date.now(),
                title: title,
                category: category,
                thumbnail: thumbnail
            };
            
            movies.push(newMovie);
            localStorage.setItem('movies', JSON.stringify(movies));
            
            document.getElementById('movieTitle').value = '';
            document.getElementById('movieThumbnail').value = '';
            
            showNotification('Thêm phim thành công!', 'success');
            renderMovies();
        }

        // Thêm danh mục (Admin)
        function addCategory() {
            if (!currentUser || !currentUser.isAdmin) {
                showNotification('Bạn không có quyền thêm danh mục!', 'error');
                return;
            }
            
            const newCat = document.getElementById('newCategory').value.trim();
            
            if (!newCat) {
                showNotification('Vui lòng nhập tên danh mục!', 'error');
                return;
            }
            
            if (categories.includes(newCat)) {
                showNotification('Danh mục đã tồn tại!', 'error');
                return;
            }
            
            categories.push(newCat);
            localStorage.setItem('categories', JSON.stringify(categories));
            
            document.getElementById('newCategory').value = '';
            updateCategorySelect();
            renderCategories();
            showNotification('Thêm danh mục thành công!', 'success');
        }

        function focusAddCategory() {
            document.getElementById('newCategory').focus();
            document.getElementById('adminPanel').style.display = 'block';
            document.getElementById('newCategory').scrollIntoView({ behavior: 'smooth' });
        }

        // Xem phim
        function playMovie(title) {
            if (!currentUser) {
                showNotification('Vui lòng đăng nhập để xem phim!', 'error');
                return;
            }
            alert(`🎬 Đang phát: ${title}\n\nChúc bạn xem phim vui vẻ!`);
        }

        // Modal functions
        function openModal(modalId) {
            document.getElementById(modalId).classList.add('active');
        }

        function closeModal(modalId) {
            document.getElementById(modalId).classList.remove('active');
        }

        // Notification
        function showNotification(message, type) {
            const notification = document.createElement('div');
            notification.className = `notification ${type}`;
            notification.textContent = message;
            document.body.appendChild(notification);
            
            setTimeout(() => {
                notification.remove();
            }, 3000);
        }

        function updateInterface() {
            renderNav();
            renderCategories();
            renderMovies();
            renderAdminPanel();
        }

        // Event listeners cho modal
        window.onclick = function(event) {
            if (event.target.classList.contains('modal')) {
                event.target.classList.remove('active');
            }
        }
